018package net.tridentsdk.util;
019
020import jdk.nashorn.internal.ir.annotations.Immutable;
021
022/**
023 * String utilities
024 *
025 * @author The TridentSDK Team
026 * @since 0.3-alpha-DP
027 */
028@Immutable
029public final class StringUtil {
030    private StringUtil() {
031    }
032
033    /**
034     * A for-loop efficient method for concating strings (or in some cases objects)
035     *
036     * @param objects Objects you wish to concat into a String
037     * @return Built string
038     */
039    public static String concat(Object... objects) {
040        StringBuilder builder = new StringBuilder();
041
042        for (Object o : objects) {
043            builder.append(o);
044        }
045
046        return builder.toString();
047    }
048
049    /**
050     * Checks if a string is a number
051     *
052     * @param str the string to check
053     * @return {@code true} is number parsable
054     */
055    public static boolean isNumeric(String str) {
056        try {
057            Integer.parseInt(str);
058        } catch (NumberFormatException ex) {
059            return false;
060        }
061
062        return true;
063    }
064}
